36 bool susceptible (Host const& h) { return std::get<0>(h) == 0; }
37 bool deceased (Host const& h) { return std::get<0>(h) == -1; }
38 bool recovered (Host const& h) { return std::get<0>(h) == -2; }
39 bool exposed (Host const& h) { return 0 < std::get<0>(h) && std::get<0>(h) <= std::get<1>(h); }
40 bool infectious (Host const& h) { return std::get<1>(h) < std::get<0>(h) && std::get<0>(h) <= (std::get<1>(h) + std::get<2>(h)); }

66int isExposed (int hi, int hj, GridMap const& m, Pathogen p)
67{
68 int count = 0;
69 int N = m.size() - 1, M = m[0].size() - 1;
70 for (auto i = std::max(0, hi - 1); i <= std::min(hi + 1, N); ++i) {
71 for (auto j = std::max(0, hj - 1); j <= std::min(hj + 1, M); ++j) {
72 if (p.infectious(m[i][j])) {
73 ++count;

82GridMap computeNext (GridMap const& m, Pathogen p)
83{
84 GridMap m_next { m };
85 int N = m.size();
86 int M = m[0].size();
87 for (auto i = 0; i < N; ++i) {
88 auto& row = m[i];
89 for (auto j = 0; j < M; ++j) {
90 auto& cell = row[j];
91 auto& cellnext = m_next[i][j];
92 if (p.exposed(cell)) {
93 ++std::get<0>(cellnext);
94 } else if (p.infectious(cell)) {
95 ++std::get<0>(cellnext);
96 auto hk = std::get<3>(cellnext);
97 for (auto hi = std::max(0, i - hk); hi <= std::min(i + hk, N - 1); ++hi) {
98 for (auto hj = std::max(0, j - hk); hj <= std::min(j + hk, M - 1); ++hj) {
99 auto& neighbor = m_next[hi][hj];
100 if (p.susceptible(neighbor)) {
101 if (p.catches()) {
102 std::get<1>(neighbor) = p.incubates();
103 std::get<2>(neighbor) = p.infects();
104 ++std::get<0>(neighbor);
105 }
106 }
107 }
108 }
109
110 if (!p.infectious(cellnext)) {
111 if (p.dies()) {
112 std::get<0>(cellnext) = -1;
113 } else {
114 std::get<0>(cellnext) = -2;
115 }
116 }
117 }
118 }
119 }
120 return m_next;
121}

146void seedDisease (GridMap& m, Pathogen p, int count) {
147 std::random_device rd;
148 //std::mt19937 gen(rd());
149 std::default_random_engine gen(rd());
150 std::uniform_int_distribution<> d(0, (m.size() * m[0].size()) - 1);
151 while (count--) {
152 auto k = d(gen);
153 auto i = k / m.size();
154 auto j = k % m.size();
155 auto& cell = m[i][j];
156 std::get<1>(cell) = p.incubates();
157 std::get<2>(cell) = p.infects();
158 std::get<0>(cell) = 1;
159 }
